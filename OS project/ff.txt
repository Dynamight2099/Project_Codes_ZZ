void allocateBandwidth(WiFi_Profile *wifi_profiles, int num_profiles, int total_bandwidth) {
    int current_bandwidth = total_bandwidth;
    int bandwidth_per_device = total_bandwidth / num_profiles;

    for (int i = 0; i < num_profiles; i++) {
        wifi_profiles[i].allocated_bandwidth = bandwidth_per_device;
        current_bandwidth -= bandwidth_per_device;
    }

    // Distribute any remaining bandwidth in a round-robin fashion
    int index = 0;
    while (current_bandwidth > 0) {
        wifi_profiles[index].allocated_bandwidth++;
        current_bandwidth--;
        index = (index + 1) % num_profiles;
    }
}
